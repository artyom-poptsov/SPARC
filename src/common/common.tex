%% This procedure allows to insert a music note.
%% Syntax:
%%   \musicnote{<octave>}{<note-name>}{<frequency>}
\newcommand{\musicnote}[4]{
  &
  \ifstrequal{#3}{C}{
    \ifstrequal{#1}{en}{Do}{До}   & C#2
  }{}
  \ifstrequal{#3}{D}{
    \ifstrequal{#1}{en}{Re}{Ре}   & D#2
  }{}
  \ifstrequal{#3}{E}{
    \ifstrequal{#1}{en}{Mi}{Ми}   & E#2
  }{}
  \ifstrequal{#3}{F}{
    \ifstrequal{#1}{en}{Fa}{Фа}   & F#2
  }{}
  \ifstrequal{#3}{G}{
    \ifstrequal{#1}{en}{Sol}{Соль} & G#2
  }{}
  \ifstrequal{#3}{A}{
    \ifstrequal{#1}{en}{La}{Ля}   & A#2
  }{}
  \ifstrequal{#3}{B}{
    \ifstrequal{#1}{en}{Si}{Си}   & B#2 (H#2)
  }{}
  & #4 \\
}

%% Taken from:

\newcommand{\textdirectcurrent}{%
  \settowidth{\dimen0}{$=$}%
  \vbox to .85ex {\offinterlineskip
    \hbox to \dimen0{\leaders\hrule\hfill}
    \vskip.35ex
    \hbox to \dimen0{%
      \leaders\hrule\hskip.2\dimen0\hfill
      \leaders\hrule\hskip.2\dimen0\hfill
      \leaders\hrule\hskip.2\dimen0
    }
    \vfill
  }%
}

\newcommand{\note}[2]{
  \def\lang{\detokenize{#1}}
  \def\langRu{\detokenize{ru}}
  \def\langEn{\detokenize{en}}
  \def\noteText{Note}
  \ifx \lang\langRu
  \def\noteText{Примечание}
  \fi
  \vspace{8pt}
  \begin{tabularx}{\textwidth}{m{1cm} m{9cm}}
    \includesvg[width=1cm]{the-noun-project/note}
    & \textbf{\noteText:} #2 \\
  \end{tabularx}
}

}%
