    \draw[thick] (0, 0.5) -- (8, 0.5) node[anchor=north west] {t};

      \draw (\x, 0) -- (\x, 1) -- (\x, 1) node[midway, above] {\n};
    };

    \foreach \x in {0, 2, ..., 6} {
      \draw (\x, 0) -- (\x, 0.5) node[pos=0.25, right] {$ \frac{1}{4} $};

    };

}
