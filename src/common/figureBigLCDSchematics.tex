  }

    \centering
    \begin{tikzpicture}
      \draw[gray, step=0.5] (0, 0) grid (10, 2);
      \foreach [count=\i from 0] \x in {0.0, 0.5, 1.0, ..., 9.5} {

      }
      \foreach [count=\i from 0] \y in {1.5, 1.0, ..., 0.0} {
        \draw (-0.5, \y) node[anchor=south west] {\i};
      }
    \end{tikzpicture}
